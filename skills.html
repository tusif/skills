        // Function to create the pivot table

        function createPivotTable(data) {

            var pivotData = {};


            // Process the data and populate the pivotData object

            for (var i = 0; i < data.length; i++) {

                var name = data[i][0];

                var skill = data[i][1];

                var detail = data[i][2];

                var date = data[i][3];


                if (!pivotData[name]) {

                    pivotData[name] = {};

                }


                if (!pivotData[name][skill]) {

                    pivotData[name][skill] = {};

                }


                pivotData[name][skill]["Detail"] = detail;

                pivotData[name][skill]["Date"] = date;

            }


            // Generate the pivot table HTML

            var table = document.getElementById("pivotTable");


            // Create the table header

            var thead = table.createTHead();

            var headerRow = thead.insertRow();

            var nameHeader = headerRow.insertCell();

            nameHeader.textContent = "Name";

            nameHeader.rowSpan = 2;


            var skillHeaderRow = thead.insertRow();

            var skillHeaders = Object.keys(pivotData[data[0][0]]);


            for (var i = 0; i < skillHeaders.length; i++) {

                var skillHeader = skillHeaderRow.insertCell();

                skillHeader.textContent = skillHeaders[i];

            }


            // Create the table body

            var tbody = table.createTBody();


            for (var name in pivotData) {

                var nameRow = tbody.insertRow();

                var nameCell = nameRow.insertCell();

                nameCell.textContent = name;


                for (var j = 0; j < skillHeaders.length; j++) {

                    var skillCell = nameRow.insertCell();

                    var skill = skillHeaders[j];


                    if (pivotData[name][skill]) {

                        skillCell.textContent = pivotData[name][skill]["Detail"];

                        skillCell.setAttribute("data-date", pivotData[name][skill]["Date"]);

                    }

                }

            }


            // Sort the table by date (if required)

            var rows = Array.from(tbody.rows);

            rows.sort(function (a, b) {

                var aDate = new Date(a.cells[1].getAttribute("data-date"));

                var bDate = new Date(b.cells[1].getAttribute("data-date"));

                return aDate - bDate;

            });


            rows.forEach(function (row) {

                tbody.appendChild(row);

            });

        }
